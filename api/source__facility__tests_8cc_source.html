00155   // Null response
00156   EXPECT_NO_THROW(src_facility->GetMatlTrades(trades, responses));
00157   EXPECT_EQ(responses.size(), 0);
00158 
00159   double qty = capacity / 3;
00160   Request<Material>::Ptr request =
00161       Request<Material>::Create(get_mat(), &trader, commod);
00162   Bid<Material>::Ptr bid =
00163       Bid<Material>::Create(request, get_mat(), src_facility);
00164 
00165   Trade<Material> trade(request, bid, qty);
00166   trades.push_back(trade);
00167 
00168   // 1 trade
00169   ASSERT_EQ(src_facility->current_capacity(), capacity);
00170   src_facility->GetMatlTrades(trades, responses);
00171   EXPECT_EQ(responses.size(), 1);
00172   EXPECT_EQ(responses[0].second->quantity(), qty);
00173   EXPECT_EQ(responses[0].second->comp(), recipe);
00174 
00175   // 2 trades, total qty = capacity
00176   ASSERT_DOUBLE_EQ(src_facility->current_capacity(), capacity - qty);
00177   ASSERT_GT(src_facility->current_capacity() - 2 * qty, -1 * cyclus::eps());
00178   trades.push_back(trade);
00179   responses.clear();

00181   EXPECT_EQ(responses.size(), 2);
00182   ASSERT_TRUE(cyclus::AlmostEq(src_facility->current_capacity(), 0));
00183 
00184   // too much qty, capn!
00185   EXPECT_THROW(src_facility->GetMatlTrades(trades, responses),

00187   
00188   // reset!
