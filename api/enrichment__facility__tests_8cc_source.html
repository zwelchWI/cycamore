00515 TEST_F(EnrichmentFacilityTest, Response) {
00516   // this test asks the facility to respond to multiple requests for enriched
00517   // uranium. two requests are provided, whose total equals the swu capacity of
00518   // the facility while not exceeding its inventory capacity (that's taken care
00519   // of in the Enrich tests).
00520   //
00521   // note that response quantity and quality need not be tested, because they
00522   // are covered by the Enrich and Offer tests
00523   using cyclus::Bid;
00524   using cyclus::CompMap;
00525   using cyclus::Composition;
00526   using cyclus::MatQuery;
00527   using cyclus::Material;
00528   using cyclus::Material;
00529   using cyclus::Request;
00530   using cyclus::Trade;
00531   using cyclus::enrichment::Assays;
00532   using cyclus::enrichment::FeedQty;
00533   using cyclus::enrichment::SwuRequired;
00534   using cyclus::enrichment::UraniumAssay;
00535   using test_helpers::get_mat;
00536   using test_helpers::trader;
00537 
00538   // problem set up
00539   std::vector< cyclus::Trade<cyclus::Material> > trades;
00540   std::vector<std::pair<cyclus::Trade<cyclus::Material>,
00541                         cyclus::Material::Ptr> > responses;
00542 
00543   double qty = 5; // 5 kg
00544   double trade_qty = qty / 3;
00545   double product_assay = 0.05; // of 5 w/o enriched U
00546   cyclus::CompMap v;
00547   v[92235] = product_assay;
00548   v[92238] = 1 - product_assay;
00549   // target qty need not be = to request qty
00550   Material::Ptr target = cyclus::Material::CreateUntracked(
00551       qty + 10, cyclus::Composition::CreateFromMass(v)); 
00552 
00553   Assays assays(feed_assay, UraniumAssay(target), tails_assay);
00554   double swu_req = SwuRequired(qty, assays);
00555   double natu_req = FeedQty(qty, assays);
00556 
00557   src_facility->SetMaxInventorySize(natu_req * 4); // not capacitated by nat u
00558   src_facility->swu_capacity(swu_req); // swu capacitated
00559 
00560   // Null response
00561   src_facility->GetMatlTrades(trades, responses);
00562   EXPECT_NO_THROW();
00563   EXPECT_EQ(responses.size(), 0);
00564 
00565   // set up state
00566   DoAddMat(GetMat(natu_req * 2));
00567   
00568   Request<Material>::Ptr req =
00569       Request<Material>::Create(target, &trader, out_commod);
00570   Bid<Material>::Ptr bid = Bid<Material>::Create(req, target, src_facility);
00571   Trade<Material> trade(req, bid, trade_qty);
00572   trades.push_back(trade);
00573   
00574   // 1 trade, SWU < SWU cap
00575   ASSERT_DOUBLE_EQ(src_facility->current_swu_capacity(), swu_req);
00576   src_facility->GetMatlTrades(trades, responses);
00577   ASSERT_EQ(responses.size(), 1);
00578   EXPECT_DOUBLE_EQ(src_facility->current_swu_capacity(),
00579                    swu_req - SwuRequired(trade_qty, assays));
00580 
00581   // 2 trades, SWU = SWU cap
00582   ASSERT_GT(src_facility->current_swu_capacity() - 2 * swu_req / 3,
00583             -1 * cyclus::eps());
00584   trades.push_back(trade);
00585   responses.clear();
00586   EXPECT_NO_THROW(src_facility->GetMatlTrades(trades, responses));
00587   EXPECT_EQ(responses.size(), 2);
00588   EXPECT_TRUE(cyclus::AlmostEq(src_facility->current_swu_capacity(), 0));
00589   
00590   // too much qty, capn!
00591   trade = Trade<Material>(req, bid, 1); // a small number
00592   trades.clear();
00593   trades.push_back(trade);
00594   EXPECT_THROW(src_facility->GetMatlTrades(trades, responses),
00595                cyclus::ValueError);
00596   
00597   // reset!
00598   src_facility->Tick(1);
00599   EXPECT_DOUBLE_EQ(src_facility->current_swu_capacity(), swu_req);
00600 }
